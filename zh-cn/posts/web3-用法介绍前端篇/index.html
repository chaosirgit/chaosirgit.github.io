this.web3.eth.sendTransaction({
    from: accounts[0], //转出地址
    to: Address,	//转入地址
    value: new Decimal(0.12).mul(new Decimal(this.bnbDecimal)).toNumber().toFixed(0), // IntString
}).on('transactionHash',  (hash) => {
  			//交易发出并得到有效的交易HASH-返回 hash
  			console.log(hash) //String
}).on('receipt', (receipt) => {
  			//收到交易收据后立刻触发
				console.log(receipt) //Object-收据对象
}).on('confirmation', (confirmationNumber, receipt) => {
				//每次链上确认后立刻触发-最多确认12次
  			console.log(confirmationNumber) //Int-0号确认开始
  			console.log(receipt) //Object-收据对象
}).on('error', (err) => {
				//交易过程中出现错误立即触发，如果交易被网络拒绝时附带有交易收据,则 err 为 交易收据
  			console.log(err)
});

function generate_wallet(){
        $config = [
            'private_key_type' => OPENSSL_KEYTYPE_EC,
            'curve_name' => 'secp256k1'
        ];
        $res = openssl_pkey_new($config);
        if (!$res) {
            echo 'ERROR: Fail to generate private key. -> ' . openssl_error_string();
            exit;
        }
        // 生成私钥
        openssl_pkey_export($res, $priv_key);
// 获取公钥
        $key_detail = openssl_pkey_get_details($res);
        $pub_key = $key_detail["key"];
        $priv_pem = \Sop\CryptoEncoding\PEM::fromString($priv_key);
// 转换为椭圆曲线私钥格式
        $ec_priv_key = \Sop\CryptoTypes\Asymmetric\EC\ECPrivateKey::fromPEM($priv_pem);
// 然后将其转换为ASN1结构
        $ec_priv_seq = $ec_priv_key->toASN1();
// HEX中的私钥和公钥
        $priv_key_hex = bin2hex($ec_priv_seq->at(1)->asOctetString()->string());
        $priv_key_len = strlen($priv_key_hex) / 2;
        $pub_key_hex = bin2hex($ec_priv_seq->at(3)->asTagged()->asExplicit()->asBitString()->string());
        $pub_key_len = strlen($pub_key_hex) / 2;
// 从公钥导出以太坊地址
// 每个EC公钥始终以0x04开头，
// 我们需要删除前导0x04才能正确hash它
        $pub_key_hex_2 = substr($pub_key_hex, 2);
        $pub_key_len_2 = strlen($pub_key_hex_2) / 2;
// Hash
        $hash = \kornrunner\Keccak::hash(hex2bin($pub_key_hex_2), 256);
// 以太坊地址长度为20个字节。 （40个十六进制字符长）
// 我们只需要最后20个字节作为以太坊地址
        $wallet_address = '0x' . substr($hash, -40);
        $wallet_private_key = '0x' . $priv_key_hex;
        return [$wallet_address,$wallet_private_key];
    }

//wei 转 10 进制
function wei_to_dec($wei,$decimal = 4,$unit = 'ether'){
        list($bnq, $bnr) = \Web3\Utils::fromWei($wei, $unit);
        $bnr = str_pad($bnr->toString(),18,'0',STR_PAD_LEFT);
        $bnr_string = mb_substr($bnr,0,$decimal);
//				需要四舍五入打开注释
//        $is_cd = mb_substr($bnr->toString(),$decimal,1);
//        if ($is_cd > 4){
//            $bnr_string = bcadd($bnr_string,1,0);
//        }
        return (string) $bnq->toString() . '.' . $bnr_string;
}
//对 ABI 方法签名
function set_abi_function_name_signature($abi,$path){
        $object = json_decode($abi);
        if (!is_array($object)){
            return false;
        }
        $contract = get_contract($abi);
        $results = collect($object)->map(function($item) use ($contract){
            if ($item->type == 'function'){
                if (empty($item->signature)){
                    $item->signature = $contract->getEthabi()->encodeFunctionSignature($item);
                }
            }
            return $item;
        });
        \Illuminate\Support\Facades\File::put($path,json_encode($results));
}
//获取交易信息
function get_transaction_data_by_hash($hash){
        $data = [];
        $data['error'] = null;
        $eth = get_eth();
        $eth->getTransactionReceipt($hash,function($err,$res) use (&$data){
            if ($err == null){
                $data['status'] = hexdec($res->status);
            }else{
                $data['error'] = $err;
            }
        });
        $eth->getTransactionByHash($hash,function($err,$res) use (&$data) {
            if ($err == null){
                $data['blockNumber'] = \Web3\Utils::toString(hexdec($res->blockNumber));
                $data['blockHash'] = $res->blockHash;
                $data['from'] = $res->from;
                $data['to'] = $res->to;
                $data['gas'] = number_format(hexdec($res->gas),0,'','');
                $data['gasPrice'] = number_format(hexdec($res->gasPrice),0,'','');
                $data['transactionIndex'] = \Web3\Utils::toString(hexdec($res->transactionIndex));
                $data['value'] = number_format(hexdec($res->value),0,'','');
                $data['input'] = $res->input;
                $eth2 = get_eth();
                $eth2->getBlockByHash($data['blockHash'],false,function($e,$d) use (&$data) {
                    if ($e == null){
                        $data['timeStamp'] = \Web3\Utils::toString(hexdec($d->timestamp));
                    }else{
                        $data['error'] = $e;
                    }
                });
            }else{
                $data['error'] = $err;
            }
        });
        return $data;
}
//解析 Input hash 数据
function input_decode($abi,$hex_string){
        $contract = get_contract($abi);
        $method_sign = substr($hex_string,0,10);
        $method_params = substr($hex_string,10);
        $abi_collect = collect($contract->getAbi());
        $method = $abi_collect->filter(fn($item) => isset($item['signature']) && ($item['signature'] == $method_sign)
        )->values()->all();
        $method_name = '';
        $params_data = [];
        if (!empty($method)){
            foreach ($method as $item){
                $method_name = $item['name'];
                if (!empty($item['inputs'])){
                    $inputs = array_map(fn($row) => $row['type'],$item['inputs']);
                    $inputs_name = array_map(fn($row) => $row['name'],$item['inputs']);
                    $inputs_data = array_map(fn($item) => $item instanceof \phpseclib\Math\BigInteger ? $item->toString() : $item,$contract->getEthabi()->decodeParameters($inputs,$method_params));
                    if (!empty($inputs_name) && !empty($inputs_data)){
                        $params_data = array_combine($inputs_name,$inputs_data);
                    }
                }
            }
        }

        $params_data['method_name'] = $method_name;
        return $params_data;
    }

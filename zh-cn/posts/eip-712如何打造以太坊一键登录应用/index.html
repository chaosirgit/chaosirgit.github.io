func main()  {
	td := apitypes.TypedData{
		Domain: apitypes.TypedDataDomain{
			Name: "testSign",
			ChainId: math.NewHexOrDecimal256(1),
			Version: "1",
		},
		Types: apitypes.Types{
			"EIP712Domain": {
				{Name: "name", Type: "string"},
				{Name: "version", Type: "string"},
				{Name: "chainId", Type: "uint256"},
			},
			"Mail" : []apitypes.Type{
				{Name: "email", Type: "string"},
        {Name: "key", Type: "uint256"},
			},
		},
		Message: apitypes.TypedDataMessage{
			"email": "test@test.com",
      "key": 123456
		},
		PrimaryType: "Mail",
	}

  //å‰ç«¯æäº¤çš„ç­¾å
	incomingMetamaskSignature := "0xeb09063ff......f191c"
  //å‰ç«¯æäº¤çš„é’±åŒ…åœ°å€
  walletAddress := "0x0296c8d68d3dd2df631F740BB13C2583Ca4741B0"
  
	typedDataHash,_ := td.HashStruct(td.PrimaryType,td.Message)
	domainSeparator, _ := td.HashStruct("EIP712Domain", td.Domain.Map())
	rawData := []byte(fmt.Sprintf("\x19\x01%s%s", string(domainSeparator), string(typedDataHash)))
	dataHash := crypto.Keccak256Hash(rawData)

	signature := hexutil.MustDecode(incomingMetamaskSignature)
	if len(signature) != 65 {
		fmt.Println("invalid signature length: %d", len(signature))
	}
	if signature[64] != 27 && signature[64] != 28 {
		fmt.Println("invalid recovery id: %d", signature[64])
	}
	signature[64] -= 27
	pubKey,err := crypto.SigToPub(dataHash.Bytes(), signature)
	if err != nil {
		fmt.Println("invalid signature: %s", err.Error())
	}
	recoveredAddr := crypto.PubkeyToAddress(*pubKey)
	fmt.Println(recoveredAddr.String())
	if recoveredAddr.String() == walletAddress {
		//Success
    fmt.Println("verify success!")
    
	}
  fmt.Println("verify error!")
}

H5 Spine 动画运行库

说明

运行库版本要与 Spine 的版本匹配

使用 PIXI 第5版 与 pixi-spine 2.1.14 运行 Spine 3.8

安装 PIXI V5 

1

npm i pixi.js@5.3.11

安装 PIXI-SPINE 

1

npm i pixi-spine@2.1.14

引入

引入 pixi.js 
1

import * as PIXI from "pixi.js"
引入 pixi-spine 给 PIXI 丰富方法 
1
2

window.PIXI = PIXI
import("pixi-spine")

使用

HTML 
1

  <div id="pixi2"></div>

 function runSpine() {
        const app = new PIXI.Application({
          width: 800,         //default:800
          height: 600,        //default:600
          antialias: true,    //default:false 字体的边界和图形更加平滑
          transparent: false, //default:false 设置整个canvas元素的透明度
          resolution: 1,       //default:1
          forceCanvas: true //使用Canvas来取代webGL渲染
        });
        app.renderer.backgroundColor = 0xFFFFFF; //背景颜色
        // 将创建好的canvas添加到元素当中去
        document.getElementById("pixi2").appendChild(app.view);
        // TODO 路径暂时写死
        app.loader
            .add('test', 'https://api.test.com/storage/pt/no1.json')
            .load(function (loader, resources) {
              console.log(resources)
              const animation = new PIXI.spine.Spine(resources.test.spineData);
              animation.x = 400	//渲染对象在画布中的 x 坐标
              animation.y = 400 //渲染对象在画布中的 y 坐标
              animation.width = 300 //渲染对象的宽
              animation.height = 300 //渲染对象的高
              // add the animation to the scene and render...
              app.stage.addChild(animation);

              if (animation.state.hasAnimation('action/idle')) {
                // 设置动画
                animation.state.setAnimation(0, 'action/idle', true);
              }
          		//运行动画
              app.start();
            });
      }

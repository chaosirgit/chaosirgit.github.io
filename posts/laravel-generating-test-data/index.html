$factory->define(App\User::class, function (Faker\Generator $faker) {
    $faker = Faker\Factory::create('zh_CN'); //chinese package
    return [
        'openid'       => str_random(10),
        'nickname'     => $faker->name,  //Chinese name
        'mobile'       => $faker->phoneNumber,
        'avatar'       => $faker->imageUrl(),
        'integral'     => $faker->randomNumber(3),
        'balance'      => $faker->randomFloat(2, 0, 10000), // Random floating point number, 2 decimal places, min 0, max 10000
        'birthday'     => $faker->date(), 
        'created_time' => $faker->unixTime(), 
        'password'     => App\User::generatePassword('haha123'), // Available model methods to generate data
    ];
});

$factory->define(App\Product::class, function (Faker\Generator $faker) {
    $faker  = Faker\Factory::create('zh_CN');
    // Use the model to generate the array to be associated with the random
    $seller = App\Seller::where('status', 1)->get()->toArray();
    foreach ($seller as $value) {
        $row[] = $value['id'];
    }
    return [
        'product_no'   => $faker->randomNumber(8),
        'name'         => 'ã€Test Productsã€‘' . $faker->streetName,
        'cate_id'      => random_int(1, 6),
        'seller_id'    => $faker->randomElement($row),
        'mod_id'       => random_int(1, 10),
        'price'        => $faker->randomFloat(2, 1, 500),
        'weight'       => random_int(1, 100),
        'img'          => $faker->imageUrl('480','480'),
        'thumb'        => $faker->imageUrl('480','480'),
        'content'      => 'ã€Test product contentã€‘' . $faker->text(),
        'is_point'     => random_int(0, 1),
        'is_new'       => random_int(0, 1),
        'is_recommend' => random_int(0, 1),
        'is_sale'      => random_int(0, 1),
        'create_time'  => $faker->unixTime(),
    ];
});

$factory->define(App\Seller::class, function (Faker\Generator $faker) {
    $faker = Faker\Factory::create('zh_CN');
    $user  = App\User::all()->toArray();
    foreach ($user as $value) {
        $row[] = $value['id'];
    }
    return [
        'user_id'     => $faker->randomElement($row),
        'seller_name' => $faker->colorName,  // Very meaningful Chinese color names
        'corporate'   => $faker->imageUrl(),
        'business'    => $faker->imageUrl(),
        'create_time' => $faker->unixTime(),
    ];
});

Preface

Authorized DAPP login using Ethereum EIP-712 protocol signature

Design

  1. the front end needs to authorize the application to connect to the wallet before using the signature
  2. get the wallet address after successful authorization
  3. then click the signature button
  4. use the wallet address to request the back-end get random number API
  5. the backend generates a random number, puts the wallet address as the key and the random number as the value into the cache, with an expiration time of 5 minutes (security considerations)
  6. the front-end write the random number into the signature message body
  7. call the wallet signature
  8. the obtained signature requests the back-end login API
  9. the backend decrypts the signature
  10. Verify that the decrypted wallet address is the same as the submitted address
  11. Register or login

The back-end get random number API is not documented in this document

Step

Connect Wallet

Authorization to connect to the wallet is required prior to signing.

func main()  {
	td := apitypes.TypedData{
		Domain: apitypes.TypedDataDomain{
			Name: "testSign",
			ChainId: math.NewHexOrDecimal256(1),
			Version: "1",
		},
		Types: apitypes.Types{
			"EIP712Domain": {
				{Name: "name", Type: "string"},
				{Name: "version", Type: "string"},
				{Name: "chainId", Type: "uint256"},
			},
			"Mail" : []apitypes.Type{
				{Name: "email", Type: "string"},
        {Name: "key", Type: "uint256"},
			},
		},
		Message: apitypes.TypedDataMessage{
			"email": "test@test.com",
      "key": 123456
		},
		PrimaryType: "Mail",
	}

  // Signature submitted by the front-end
	incomingMetamaskSignature := "0xeb09063ff......f191c"
  // Wallet address by the front-end
  walletAddress := "0x0296c8d68d3dd2df631F740BB13C2583Ca4741B0"
  
	typedDataHash,_ := td.HashStruct(td.PrimaryType,td.Message)
	domainSeparator, _ := td.HashStruct("EIP712Domain", td.Domain.Map())
	rawData := []byte(fmt.Sprintf("\x19\x01%s%s", string(domainSeparator), string(typedDataHash)))
  // If your sign method is personal_sign then sign data must add prefix like this:
  // rawData := []byte(fmt.Sprintf("\x19Ethereum Signed Message:\n%d%s",len("Hello "+"123456"),"Hello 123456"))
  dataHash := crypto.Keccak256Hash(rawData)

	signature := hexutil.MustDecode(incomingMetamaskSignature)
	if len(signature) != 65 {
		fmt.Println("invalid signature length: %d", len(signature))
	}
	if signature[64] != 27 && signature[64] != 28 {
		fmt.Println("invalid recovery id: %d", signature[64])
	}
	signature[64] -= 27
	pubKey,err := crypto.SigToPub(dataHash.Bytes(), signature)
	if err != nil {
		fmt.Println("invalid signature: %s", err.Error())
	}
	recoveredAddr := crypto.PubkeyToAddress(*pubKey)
	fmt.Println(recoveredAddr.String())
	if recoveredAddr.String() == walletAddress {
		//Success
    fmt.Println("verify success!")
    
	}
  fmt.Println("verify error!")
}

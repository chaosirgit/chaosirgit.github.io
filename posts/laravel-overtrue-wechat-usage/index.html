    public function handle($request, Closure $next, $account = 'default', $scopes = null)
        {
            // The case where $account and $scopes are written backwards
            if (is_array($scopes) || (\is_string($account) && str_is('snsapi_*', $account))) {
                list($account, $scopes) = [$scopes, $account];
                $account || $account = 'default';
            }
    
            $isNewSession = false;
            $sessionKey = \sprintf('user_id', $account);
            $config = config(\sprintf('wechat.official_account.%s', $account), []);
            $officialAccount = app(\sprintf('wechat.official_account.%s', $account));
            $scopes = $scopes ?: array_get($config, 'oauth.scopes', ['snsapi_base']);
    
            if (is_string($scopes)) {
                $scopes = array_map('trim', explode(',', $scopes));
            }
    
    
            //$session = session([$sessionKey=>'24260']); // Test environment directly assign session Formal environment comments
            
               /**
                * 1. whether there is a session
                * 2. no session determine if there is a code
                * 3. no code get-code-callback to this page
                * 4. weibo return code parameter to request this page
                * 5. with code get openid to perform database operation * 6.
                * 6. openid is in the database to assign session to jump to the next request
                * 7. openid is no longer in the database, determine if the nickname can be fetched, if not, use snsapi_userinfo to authorize login, otherwise, store the relevant information in the database and assign the session
                */
            
            if (!session($sessionKey)) {                          //1. No session
                if ($request->has('code')) {                 //2. Have code
    
                    $wechat_user = $officialAccount->oauth->user();         
                    $data['openid'] = $wechat_user->getId() ?? '';          //get openid
                    $has = UserWechat::getByOpenid($data['openid']);
                    $has_user = User::where('tync_openid',$data['openid'])->first();
                    if(!empty($has) && !empty($has_user)){              //openid Jump to the next step in the library
                        $user_id = User::where('tync_openid',$data['openid'])->first()->id ?? 0;
                        session([$sessionKey => $user_id ?? '']);
                        return $next($request);
                    }else{                                              // Not in the database
                        $data['nickname'] = $wechat_user->getNickname() ?? '';
                        if(empty($data['nickname'])){                   // Can't get nickname to use authorized login
                            session([$sessionKey => '']);
                            return $officialAccount->oauth->scopes(['snsapi_userinfo'])->redirect($request->fullUrl());
                        }else{                                          //Fetch the nickname and store it in the database
                            $data['headimgurl'] = $wechat_user->getAvatar() ?? '';
                            UserWechat::createUser($data);                
                            User::createWeChat($data);
                            $user_id = User::where('tync_openid',$data['openid'])->first()->id ?? 0;
                            session([$sessionKey => $user_id ?? '']);
                        }
                    }
    
                    $isNewSession = true;
    
    //                Event::fire(new WeChatUserAuthorized(session($sessionKey), $isNewSession, $account));
    
                    return redirect()->to($this->getTargetUrl($request));
                }
                /**
                 * 3.No code? Get code-callback to this page
                 */
    
                session()->forget($sessionKey);
    
                return $officialAccount->oauth->scopes($scopes)->redirect($request->fullUrl()); //Get code-callback to this page
            }
    
    //        Event::fire(new WeChatUserAuthorized(session($sessionKey), $isNewSession, $account));
            
            return $next($request);
        }

public function payment(Request $request)
    {
        $type = Input::get("type");//Payment Type
        $order_no = Input::get("order_no");//Order number
        if (!in_array($type,array("wechat_pay","amount","ali_pay","alipay_app"))){
            return $this->error("Payment type error");
        }
        $is_wechat = Input::get("is_wechat","0");// Is it a WeChat browser
        if (empty($order_no)) return $this->error("Params error");
        $order = FarmOrder::where('order_no',$order_no)->where("user_id",User::getUserId())->where("pay_status",0)->first();
        if (empty($order)) return $this->error("Order not found");
        $user_id = User::getUserId();

            //WeChat Pay
            if($type == "wechat_pay"){
            try {
                $order->pay_id = 2;
                $user          = User::find($user_id);
                $app           = app('wechat.payment.default');
                //Consolidated order (prepayment)
                $result        = $app->order->unify([
                    'body'         => 'Field Farm-Pay Order',
                    'out_trade_no' => $order_no,
                    'total_fee'    => $order->total_price * 100,
//                'spbill_create_ip' => '123.12.12.123', // Optional, if this parameter is not passed, the SDK will automatically obtain the corresponding IP address
                    'notify_url'   => 'http://www.tianyuannongchang.cn/wechat/notify', // Payment result notification URL, if not set then the default address in the configuration will be used
                    'trade_type'   => 'JSAPI',
                    'openid'       => $user->tync_openid,
                ]);
                // Generate payment jssdk based on prepayment data
                $payment = Factory::payment(config('wechat.payment.default'));
                $jssdk = $payment->jssdk;
                $json = $jssdk->bridgeConfig($result['prepay_id'],false); // return a json string, if you want to return an array, pass the second parameter false
                return $this->success($json);

            }catch (\Exception $exception){
                return $this->error($exception->getMessage());
            }
        }else{
            return $this->error("Payment type error");
        }
    }

 public function notifyWechat(Request $request)
    {

        $app = app('wechat.payment.default');
        $response = $app->handlePaidNotify(function($message, $fail){
            // Use the "WeChat Order Number" or "Merchant Order Number" in the notification to find the order in your database
            $order = FarmOrder::getOrderByNo($message['out_trade_no']);

            if (empty($order) || $order->pay_status == 1) { // If the order does not exist or if the order has already been paid for
                return true; // Tell WeChat I'm done processing, order not found, don't notify me again
            }


            if ($message['return_code'] === 'SUCCESS') { // return_code indicates the communication status, not the payment status
                // Whether the user paid successfully or not
                if ($message['result_code'] === 'SUCCESS') {
                    $order->pay_time = time(); // Update the payment time to the current time
                    $order->pay_status = 1;

                    // User payment failure
                } elseif ($message['result_code'] === 'FAIL') {
                    $order->pay_status = 0;
                }
            } else {
                return $fail('通信失败，请稍后再通知我');
            }

            $order->save(); // Save order

            return true; // Return to processing completion
        });

        return $response;

    }

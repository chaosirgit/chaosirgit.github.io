Abi, err := os.Open("/Abi.json")
if err != nil {
	 panic(err)
}
defer Abi.Close()
abiByte, _ := ioutil.ReadAll(Abi)
ContractAddress := "0x.............FF" 
contract, err := Web3.Eth.NewContract(string(abiByte), ContractAddress)
// The difference with sending the native method is the need to add a data field, data needs to be converted to inputData HEX
var amount int = 1
toAddressString := "0x.............FF" 
toAddress := common.HexToAddress(toAddressString)
amountWei := Web3.Utils.ToWei(float64(amount))
inputData := contract.EncodeABI("transfer", toAddress, amount)
call := &types.CallMsg{
		From: Web3.Eth.Address(),
		To:   toAddress,
		Data: inputData,
}
//send transaction - get Gaslimit
gasLimit, err := Web3.Eth.EstimateGas(call)
if err != nil {
			panic(err)
}
//send transaction
txHash, err := Web3.Eth.SyncSendRawTransaction(
				common.HexToAddress(ContractAddress),
				big.NewInt(0),
				gasLimit,
				Web3.Utils.ToGWei(5),
				inputData,
)
if err != nil {
	 panic(err)
}
fmt.Printf("[%s]Success tx hash: %v\n", time.Now().Format("2006-01-02 15:04:05"), txHash.TxHash)

H5 Spine Animation Runtime Library

Description

The runtime library version should match the version of Spine

Running Spine 3.8 with PIXI version 5 and pixi-spine 2.1.14

Installing Pixi V5 

1

npm i pixi.js@5.3.11

Installation of Pixi-Spine 

1

npm i pixi-spine@2.1.14

Import

Import pixi.js 
1

import * as PIXI from "pixi.js"
Import pixi-spine to PIXI enrichment methods 
1
2

window.PIXI = PIXI
import("pixi-spine")

 function runSpine() {
        const app = new PIXI.Application({
          width: 800,         //default:800
          height: 600,        //default:600
          antialias: true,    //default:false Smoother font borders and graphics
          transparent: false, //default:false Set the transparency of the entire canvas element
          resolution: 1,       //default:1
          forceCanvas: true //Using Canvas to replace webGL rendering
        });
        app.renderer.backgroundColor = 0xFFFFFF; 
        // Add the created canvas to the element
        document.getElementById("pixi2").appendChild(app.view);
        // TODO 
        app.loader
            .add('test', 'https://api.test.com/storage/pt/no1.json')
            .load(function (loader, resources) {
              console.log(resources)
              const animation = new PIXI.spine.Spine(resources.test.spineData);
              animation.x = 400	//Render the x-coordinate of the object in the canvas
              animation.y = 400 //Render the y-coordinate of the object in the canvas
              animation.width = 300
              animation.height = 300
              // add the animation to the scene and render...
              app.stage.addChild(animation);

              if (animation.state.hasAnimation('action/idle')) {
                // Set the animation
                animation.state.setAnimation(0, 'action/idle', true);
              }
          		// Run animation
              app.start();
            });
      }

public static function tree($data,$pid=0,$level=0){
        $results = array();
        foreach ($data as $value){
        // recursion point If the parent id of the current record is equal to the incoming parent id, then this record is a child of the incoming parent id
            if($value['p_id'] == $pid){
                $value['level'] = $level;
                // recursive call to get the sublevels under the sublevel
                $value['children'] = self::tree($data,$value['id'],$level + 1);
                //push assigns the above result to the returned data
                $results[] = $value;
            }
            // Recursive exit: traversal complete.
        }
        // Return the results
        return $results;
    }

    value: new Decimal(0.12).mul(new Decimal(this.bnbDecimal)).toNumber().toFixed(0), // IntString
}).on('transactionHash',  (hash) => {
  			// transaction is issued and a valid transaction HASH is obtained - return hash
  			console.log(hash) //String
}).on('receipt', (receipt) => {
  			// Triggered immediately upon receipt of transaction receipt
				console.log(receipt) //Object-Receipt object
}).on('confirmation', (confirmationNumber, receipt) => {
				// Triggered immediately after each in-chain confirmation - up to 12 confirmations
  			console.log(confirmationNumber) //Int-0 confirmation start
  			console.log(receipt) //Object-Receipt object
}).on('error', (err) => {
				// Triggered immediately by an error during the transaction, if the transaction is rejected by the network with a transaction receipt, then err is the transaction receipt
  			console.log(err)
});

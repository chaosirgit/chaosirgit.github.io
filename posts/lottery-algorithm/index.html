<?php

namespace App\Utils;


class PROBABILITY
{
    //chance is the probability of drawing, the smaller the value, the smaller the probability
// $arr = [
// ['name'=>'$1000 phone','chance'=>'1'],
// ['name'=>'$100 voucher','chance'=>'10'],
// ['name'=>'$10 voucher','chance'=>'100'],
// ['name'=>'Thanks for participating','chance'=>'700'],
// ];
    /**
     * The lottery probability algorithm
     * @param array $prizeArr array of prizes
     * @param string $index Probability key name
     * @return mixed the prizes drawn
     */
    public static function lottery_raffle($prizeArr,$index='chance')
    {
        $result = '';
        // array set to a collection
        $prize = collect($prizeArr);

        $randArr = $prize->pluck($index)->toArray();

        // total probability precision of the probability array
        $proSum = array_sum($randArr);
        //probability array loop
        foreach ($randArr as $key => $val) {
            //generate a random number - from 1 to probability and generate a random send
            $randNum = mt_rand(1, $proSum);
            //hit - if the random send is less than this probability value
            if ($randNum <= $val) {
// echo 'yes:'. $randNum.'-'. $val;
                // return the result of this probability
                $result = $key;
                break;
                //missed-total probability value is self-subtracted from this probability value
            } else {

                //probability
                $proSum -= $val;
// echo 'no:'. $randNum.'-'. $val.'Total probability updated to:'. $proSum.'|';
            }
        }
        return $prizeArr[$result];
    }
}

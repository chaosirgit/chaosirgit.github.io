    public function upload(){
        $accessKey = Setting::getValueByKey('qn_accessKey');
        $secretKey = Setting::getValueByKey('qn_secretKey');
        $bucket = Setting::getValueByKey('qn_bucket_static');
        $baseUrl = Setting::getValueByKey('qn_static_url');
        
        //Build forensic objects
        $auth = new Auth($accessKey,$secretKey);
        
        //generate upload token
        $token = $auth->uploadToken($bucket);

        return $token;
    }

<?php

public function uploadSave(Request $request){
        $key = $request->get('key',null);
        $multiple = $request->get('multiple',false);
        $num = $request->get('number',1);
        if(empty($key)){
            return $this->error('Uploading image error.Please try again');
        }
        $type = 'image';
        $user_id = User::getAdminId();
        $uploads = new Uploads();
        $uploads->type = $type;
        $uploads->user_id = $user_id;
        $uploads->key = $key;
        $uploads->time = time();
        try{
            $uploads->save();
            $result = Setting::getValueByKey('qn_static_url').'/'.$key;
            return $this->success($result);
        }catch (\Exception $exception){
            return $this->error($exception->getMessage());
        }
    }

function layer_show(title,url,w,h) {
            var width = w || null;
            var height = h || null;
            var areaValue;
            if (width != null) {
                areaValue = width + 'px';
                if (height != null) {
                    areaValue = [width + 'px', height + 'px'];
                }
            }else{
                areaValue = ['100%','100%'];
            }
                layui.use('layer', function () { // The standalone version of layer does not need to execute this sentence
                    var $ = layui.jquery, layer = layui.layer; // The standalone version of layer does not need to execute this sentence
                    layer.open({
                        type: 2 // Here is an example of an iframe
                        , title: title
                        , area: areaValue
                        , shade: 0
                        , maxmin: true
                        , content: url
                        , offset: '10px'

                    });
                });
        }

//number The maximum number of images that can be selected for multiple uploads
 function uploadInsts(elem,multiple,number){
            if(multiple == null){
                multiple = false;
            }
            if(number == null){
                number = 0;
            }
            layui.use(['upload'],function () {
                var $ = layui.jquery
                    ,upload = layui.upload;
                upload.render({

                    elem: elem
                    ,url: 'https://up.qbox.me/'         //Qiniu upload interface
                    ,data:{'token':'{{$token ?? null}}'}    //With upload token
                    ,multiple:multiple
                    ,number:number
                    // ,before: function(obj){
                    //     //Pre-read local file example, ie8 not supported
                    //     obj.preview(function(index, file, result){
                    //         $('#demo1').attr('src', result); // Image link (base64)
                    //     });
                    // }
                    ,done: function(res){
                        // After a successful upload, Qiniu returns the key and submits it to the back-end interface to save the image address and other information.
                        $.ajax({
                            url:"{{url('admin/upload_save')}}"      
                            ,data:{'key':res.key,'multiple':multiple,'num':number}
                            ,type:'post'
                            ,success:function(result){
                                if(result.error > 0){
                                    layer.msg(result.msg);
                                }else{                      // Reload this page after successful back-end save
                                    location.reload();      
                                }
                            }
                        });
                        // Upload successful
                    }
                    ,error: function(){
                        layer.msg('Please refresh and try again');
                        return false;
                    }
                });
            });
        }

// Click OK to execute the callback function name passed in from the parent page and close this page
        function backParent() {
            layui.use('form', function () {
                var $ = layui.jquery
                    , form = layui.form;

                // Note: parent is a global object that comes with JS and can be used to manipulate the parent page
                var index = parent.layer.getFrameIndex(window.name); //Get window index

                // Listening for commits
                form.on('submit(demo1)', function (data) {
                    // Passing values to the parent page
                    var s = 0;
                    var res = new Array();
                    for (var i in data.field) {
                        res.push(data.field['like[' + s + ']']);
                        s++;
                    }
                    // Pass the constructed data into the callback function
                    parent.{{$callback}}(res);      // Call the name of the callback function passed in from the parent page, where the callback function name is returned from php.
                    parent.layer.close(index);      // Close this page

                    return false;
                });
            });
        }
